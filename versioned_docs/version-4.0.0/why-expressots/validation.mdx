# Validation Comparison

Compare ExpressoTS's validation system with other major frameworks.

### ExpressoTS Advantage

- ✅ **Zero-config smart validation** - Works out of the box
- ✅ **Whitelist mode** - Automatically strips unknown properties
- ✅ **Transform mode** - Automatically converts types
- ✅ **TypeScript inference** - Type-safe validation with DTOs

### Validation Setup

ExpressoTS uses `class-validator` for validation with smart defaults:

```typescript
// Enable validation globally in App
export class App extends AppExpress {
    async configureServices(): Promise<void> {
        this.Middleware.addValidation();
    }
}
```

### DTO Validation Example

```typescript
import { IsString, IsEmail, MinLength, IsOptional, ValidateNested } from "class-validator";
import { Type } from "class-transformer";

class AddressDTO {
    @IsString()
    street: string;

    @IsString()
    city: string;
}

class CreateUserDTO {
    @IsString()
    @MinLength(2)
    name: string;

    @IsEmail()
    email: string;

    @IsString()
    @MinLength(8)
    password: string;

    @IsOptional()
    @ValidateNested()
    @Type(() => AddressDTO)
    address?: AddressDTO;
}

@controller("/users")
export class UserController {
    @Post("/")
    createUser(@body() dto: CreateUserDTO) {
        // dto is automatically validated
        // Invalid requests return 400 with detailed errors
        return { user: dto };
    }
}
```

### Validation Error Response

When validation fails, ExpressoTS returns a structured error response:

```json
{
    "type": "https://httpstatuses.io/400",
    "title": "Validation Failed",
    "status": 400,
    "detail": "Request validation failed",
    "errors": [
        {
            "property": "email",
            "constraints": {
                "isEmail": "email must be a valid email"
            }
        },
        {
            "property": "password",
            "constraints": {
                "minLength": "password must be at least 8 characters"
            }
        }
    ]
}
```

### Custom Validators

```typescript
import { registerDecorator, ValidationOptions } from "class-validator";

function IsStrongPassword(options?: ValidationOptions) {
    return function (object: Object, propertyName: string) {
        registerDecorator({
            name: "isStrongPassword",
            target: object.constructor,
            propertyName: propertyName,
            options: options,
            validator: {
                validate(value: any) {
                    const hasUpperCase = /[A-Z]/.test(value);
                    const hasLowerCase = /[a-z]/.test(value);
                    const hasNumbers = /\d/.test(value);
                    const hasSpecial = /[!@#$%^&*]/.test(value);
                    return hasUpperCase && hasLowerCase && hasNumbers && hasSpecial;
                },
                defaultMessage() {
                    return "Password must contain uppercase, lowercase, number, and special character";
                },
            },
        });
    };
}

class RegisterDTO {
    @IsStrongPassword()
    password: string;
}
```
