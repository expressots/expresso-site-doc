### Guard Caching

Unique to ExpressoTS - cache authorization results to avoid redundant checks:

```typescript
import { IGuardCache, provide, inject } from "@expressots/core";

@provide(RequestScopedGuardCache)
export class RequestScopedGuardCache implements IGuardCache {
    private cache = new Map<string, boolean>();

    has(key: string): boolean {
        return this.cache.has(key);
    }

    get(key: string): boolean | undefined {
        return this.cache.get(key);
    }

    set(key: string, value: boolean): void {
        this.cache.set(key, value);
    }
}

// Guards automatically use cache when available
@UseGuards(RequireRole("admin")) // Cached for request
@UseGuards(RequirePermission("write")) // Uses same cache
```

### Middleware Profiling

Built-in performance tracking for middleware:

```typescript
export class App extends AppExpress {
    async configureServices(): Promise<void> {
        // Enable profiling
        this.Middleware.enableProfiling();

        // Add middleware as usual
        this.Middleware.addCors();
        this.Middleware.addHelmet();
        this.Middleware.addCompression();
    }

    async postServerInitialization(): Promise<void> {
        // Get profiler stats
        const stats = this.Middleware.getProfilerStats();

        console.log("Middleware Performance:");
        stats.forEach(stat => {
            console.log(`  ${stat.name}: avg=${stat.avgTime}ms, calls=${stat.count}`);
        });
    }
}
```

### Lazy Loading Modules

Load modules on demand to reduce startup time:

```typescript
import { LazyModule } from "@expressots/core";

// Define lazy module
const AdminModule = LazyModule(() => import("./admin/admin.module"));

// Only loaded when first request hits /admin routes
@controller("/admin", AdminModule)
export class AdminController {
    // ...
}
```
